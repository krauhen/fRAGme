Docker

From dockerhub

docker pull krauhen/fragme
docker run -d \
          --name fragme \
          --hostname fragme \
          --restart always \
          -p 8080:8080 \
          -e PORT=8080 \
          -e HOST=0.0.0.0 \
          -e OPENAI_API_KEY="" \
          -e DATA_PATH="./data" \
          fragme

From source

git clone git@github.com:krauhen/fRAGme.git
cd fRAGme
docker-compose build
# set OPENAI_API_KEY in .env to "YOUR_OPENAI_API_KEY"
docker-compose up
